# p est le vecteur des probabilités correspondant à la loi de X
# x est le vecteur des valeurs possibles pour X
# M est le nombre désiré de simulation de la variable X 
def sim_dis (p, x, M):
    y = []
    for i in range(0, M):
        u = np.random.uniform(0, 1)
        j = 1
        while ( u > np.sum(p[:j]) ):
            j += 1
        y.append(x[j-1])
    return y

def simulation_marche(N=100, S0=5, p=0.5, M=1000):
    R = np.zeros((N + 1,M))
    R[0] = S0
    for n in range(1, N + 1):
        sim = sim_dis([p, 1-p], [-1, 1], M)
        R[n] = R[n - 1] + sim
    return R


def simulation_visuelle_trajectoire(N2= 100, M2=10):
    arr = simulation_marche(M=M2)
    for i in range(0, M2):
        plt.plot(np.linspace(0, N2, N2+1), arr[:,i])
    plt.show()

Détermination de l'intervalle de confiance¶

Pour déterminer l'intervalle de confiance à 95% avec $M$ variables aléatoires iid vérifiant $S_N = S_0 + X_1 + X_2 + .... + X_N$ où $\mathbb{P}(X_i = 1) = p$, $\mathbb{P}(X_i = -1) = 1-p$ avec $p \in ]0; 1[$, on cherche $a_n$ et $b_n$ tels que $\forall p \in \{-1; 1\}, \mathbb{P} ( \theta \in [a_n; b_n]) \ge 1 - \alpha$.

  • Tout d'abord, $\mathbb{E} (S_N) = \mathbb{E} (S_0) + \mathbb{E} (X_1) + \mathbb{E} (X_2) + ... + \mathbb{E} (X_N)$ par linéarité de l'espérance.

  • Or, $\mathbb{E} (X_1) = 1 * p + (-1) * (1 - p) = 2p - 1$ et les $X_i$ étant iid, elles ont la même espérance et variance; et $S_0 \in \mathbb{R}$.

  • D'où $\mathbb{E} (S_N) = S_0 + N * (2p - 1)$.

  • De plus, $\mathbb{V} (X_1) = 1^2 * p + (-1)^2 * (1-p) - (2p - 1)^2 = (1 - p) * (4p)$.

  • D'où $\mathbb{V} (S_N) = N * ((1-p)*4p)$ (car les $X_i$ sont iid).

  • D'après l'inégalité de Bienaymé-Tchebychev et si $p = 0.5$, on a:

$$ \forall p \in \{-1; 1\}, \mathbb{P} (|S_N - \mathbb{E} (S_N)| \ge \epsilon) \le \frac{\mathbb{V} (S_N)}{(\epsilon)^2}$$$$ \equiv \mathbb{P} (|S_N - S_0| \ge \epsilon) \le \frac{N}{(\epsilon)^2}$$

Or, on cherche $\mathbb{P} (|S_N - S_0| \le \epsilon) \ge 1 - (0.05)$ Donc en posant $\alpha = 0.05 = \frac{N}{(\epsilon)^2}$, on a $\epsilon = \sqrt{\frac{N}{0.05}}$, d'où $$\mathbb{P} (|S_N - S_0| \le \sqrt{\frac{N}{0.05}}) \ge (0.05)$$ $$\equiv \mathbb{P} |-\sqrt{\frac{N}{0.05}} \le S_N - S_0 \le \sqrt{\frac{N}{0.05}} \ge (0.05)$$ $$\equiv \mathbb{P} |-\sqrt{\frac{N}{0.05}} \le S_N - S_0 \le \sqrt{\frac{N}{0.05}} \ge (0.05)$$

def Q4_chaine_markov(N=3, M=1000):
    P = np.random.random((10, 10))
    P = P/np.sum(P, axis=1)[:, None] # Normalisation des lignes, pour avoir une matrice stochastique
    Mat = np.zeros((N + 1, M), dtype=int)
    Mat[0] = 1
    for n in range(1, N + 1):
        for m in range(0, M):
            s = sim_dis(P[(Mat[n-1, m]) - 1], np.linspace(1, 10, 10, dtype=int), 1)
            Mat[n, m] = s[0]
    return Mat

def death_time(Ne=5, Ns=10, P=P7, Q=P7M, N=100):
    Nb_tour = 1
    Nb_vivant = Ne
    Mat = np.zeros((N + 1, Ne), dtype=int)
    Mons = [Ns]
    Mat[0] = 1
    while (Nb_vivant > 0 and Nb_tour < N):
        s = sim_dis(Q[(Mons[-1]) - 1], np.linspace(1, 10, 10, dtype=int), 1) # Le monstre se déplace
        Mons.append(s[0])
        for m in range(0, Ne):
            if (Mat[Nb_tour-1, m] != -1):
                s = sim_dis(P[(Mat[Nb_tour-1, m]) - 1], np.linspace(1, 10, 10, dtype=int), 1)
                Mat[Nb_tour, m] = s[0]
                if (Mat[Nb_tour, m] == Mons[-1]):
                    Mat[Nb_tour, m] = -1
            else:
                Mat[Nb_tour, m] = -1
        Nb_vivant = sum(Mat[Nb_tour] != -1)
        Nb_tour += 1
    return Nb_tour - 1 # Car le premier tour n'est pas compté, c'est l'initialisation
